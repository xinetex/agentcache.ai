from adaptive_cache import Pipeline, ClinicalPHI

pipeline = Pipeline(
    name="clinical_decision_support",
    sector="healthcare",
)

pipeline.add_source("ehr", kind="fhir", cache_ttl="5m")
pipeline.add_node(ClinicalPHI(mode="redact", threshold=0.82))
pipeline.add_cache("clinical_l1", strategy="content", ttl="15m")
pipeline.add_audit("hipaa_audit", sink="s3://phi-logs")

app = pipeline.serve(on="/v1/clinical/decision")
