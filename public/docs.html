Node Quickstart

POST /api/cache/get
POST /api/cache/set

// Get (check cache)
fetch('/api/cache/get', {
  method: 'POST', headers: { 'content-type': 'application/json', 'X-API-Key': 'ac_demo_test123' },
  body: JSON.stringify({
    provider: 'openai',
    model: 'gpt-4',
    messages: [{ role: 'user', content: 'Explain Python' }]
  })
});

// Set (store response)
fetch('/api/cache/set', {
  method: 'POST', headers: { 'content-type': 'application/json', 'X-API-Key': 'ac_demo_test123' },
  body: JSON.stringify({
    provider: 'openai',
    model: 'gpt-4',
    messages: [{ role: 'user', content: 'Explain Python' }],
    response: 'Python is a programming language...'
  })
});
