Quick Start Guide

1. Get Your API Key

Sign up at agentcache.ai/community to get your free API key instantly.

2. Install SDK

npm install agentcache-sdk

3. Initialize Client

const AgentCache = require('agentcache-sdk');

const cache = new AgentCache({
  apiKey: 'your-api-key-here',
  baseUrl: 'https://agentcache.ai/api'
});

4. Cache AI Responses

// Check cache first
const cached = await cache.get({
  provider: 'openai',
  model: 'gpt-4',
  messages: [
    { role: 'user', content: 'What is 2+2?' }
  ]
});

if (cached.hit) {
  console.log('Cache hit!', cached.response);
} else {
  // Call your LLM provider
  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages: [{ role: 'user', content: 'What is 2+2?' }]
  });
  
  // Store in cache
  await cache.set({
    provider: 'openai',
    model: 'gpt-4',
    messages: [{ role: 'user', content: 'What is 2+2?' }],
    response: response.choices[0].message.content
  });
  
  console.log('Cached for next time:', response);
}

5. Cache Tool Calls

// Weather API example
const weatherCache = await cache.tool.get({
  tool_name: 'get_weather',
  parameters: { city: 'San Francisco' }
});

if (!weatherCache.hit) {
  const weather = await fetchWeather('San Francisco');
  await cache.tool.set({
    tool_name: 'get_weather',
    parameters: { city: 'San Francisco' },
    result: weather,
    ttl: 1800 // 30 minutes
  });
}

6. Monitor Your Savings

const stats = await cache.stats();

console.log(`Cache hit rate: ${stats.hitRate}%`);
console.log(`Money saved: $${stats.costSaved}`);
console.log(`Requests cached: ${stats.totalRequests}`);
