// 1) Install
// npm i agentcache openai
import AgentCache from "agentcache";
import OpenAI from "openai";

const cache = new AgentCache({ project: "my-app", defaultTtl: "7d" });
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// 2) Wrap your call — identical prompts = instant cache hits
const prompt = "Summarize: The quick brown fox jumps over the lazy dog";
const res = await cache.fetch({
  key: ["openai:gpt-4o-mini", prompt],
  tags: ["summary", "gpt-4o-mini"],
  fn: () => openai.chat.completions.create({
    model: "gpt-4o-mini",
    messages: [{ role: "user", content: prompt }],
  }),
});

console.log(res.choices[0].message.content);
                

# 1) pip install agentcache openai
from agentcache import AgentCache
from openai import OpenAI

cache = AgentCache(project="my-app", default_ttl="7d")
client = OpenAI()

prompt = "Explain vector databases in one sentence"
res = cache.fetch(
    key=["openai:gpt-4o-mini", prompt],
    tags=["explain", "gpt-4o-mini"],
    fn=lambda: client.chat.completions.create(
        model="gpt-4o-mini",
        messages=[{"role":"user","content":prompt}]
    )
)

print(res.choices[0].message.content)
                

/* Edge runtime (Workers/Functions) */
import AgentCache from "agentcache/edge";
import { Anthropic } from "@anthropic-ai/sdk";

export default async function handler(req) {
  const cache = new AgentCache({ project: "my-app", kv: env.AGENTCACHE_KV });
  const anthropic = new Anthropic({ apiKey: env.ANTHROPIC_API_KEY });

  const user = "Write a haiku about edge caching";
  const data = await cache.fetch({
    key: ["anthropic:claude-3-haiku", user],
    ttl: "3d",
    fn: () => anthropic.messages.create({
      model: "claude-3-haiku-20240307",
      max_tokens: 128,
      messages: [{ role: "user", content: user }]
    })
  });

  return new Response(JSON.stringify(data), { headers: { "content-type": "application/json" } });
}

FAQ

How do you avoid stale cache entries?

We version keys with provider model IDs and parameters. TTLs can be set per-call, per-tag, or globally. You can also purge by tag or key via the API.

Is my data secure?

All entries are encrypted at rest with server-side keys, with optional customer-managed keys. PII scrubbing is available for sensitive workloads.

Which providers do you support?

Any HTTP-based LLM API. First-class SDK helpers for OpenAI, Anthropic, and Claude models, with generic adapters for others.

What about tools and function calls?

Tools/functions are part of the cache key. You can also attach custom key material for RAG inputs or retrieval ids.
