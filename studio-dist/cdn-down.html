Integrate in 5 Minutes

1. Install AgentCache SDK

npm install @agentcache/sdk
# or
pip install agentcache

2. Add Caching to Your AI Calls

import { AgentCache } from '@agentcache/sdk';

const cache = new AgentCache({
  apiKey: 'ac_live_your_key',
  storage: 'lyve-cloud', // Seagate Lyve backend
});

// Wrap your LLM calls
const response = await cache.get({
  provider: 'openai',
  model: 'gpt-4',
  messages: [...],
});

if (!response) {
  // Cache miss - call provider
  const llmResponse = await openai.chat.completions.create({...});
  await cache.set({ provider, model, messages }, llmResponse);
}
